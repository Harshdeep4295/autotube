## 2.3 Code Implementation (Pure HeyGen)

### `agents/avatar_agent.py` (Pure HeyGen variant)

```python
"""Avatar agent — HeyGen pure approach."""
import logging
import time
from pathlib import Path
from config import config
from agents.heygen_client import HeyGenClient

logger = logging.getLogger(__name__)

class AvatarGenerator:
    def __init__(self):
        self.heygen = HeyGenClient(api_key=config.HEYGEN_API_KEY)
        self.cache_dir = Path(config.VIDEO_CACHE_DIR) / "heygen"
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        
    def generate_pure_avatar(self, script: dict, run_id: str = None) -> str | None:
        """Generate full video with avatar narration (no B-roll)."""
        try:
            # Step 1: Structure script as single narration
            narration = self._structure_script(script)
            
            # Step 2: Check cache
            cache_key = self._get_cache_key(narration)
            cached_path = self.cache_dir / f"heygen_{cache_key}.mp4"
            if cached_path.exists() and cached_path.stat().st_size > 1_000_000:
                logger.info(f"Using cached avatar video: {cached_path.name}")
                return str(cached_path)
            
            # Step 3: Create video via HeyGen
            logger.info("Creating avatar video (pure approach)...")
            operation = self.heygen.create_video(
                avatar_id=config.HEYGEN_AVATAR_ID,
                script=narration,
                voice_params={
                    "language": "en",
                    "accent": "US",
                    "speed": 1.0
                }
            )
            
            # Step 4: Poll for completion
            logger.info(f"Waiting for HeyGen generation (operation: {operation.id})...")
            max_wait = 600  # 10 minutes
            waited = 0
            while not operation.done and waited < max_wait:
                time.sleep(15)
                waited += 15
                operation = self.heygen.get_video_status(operation.id)
                logger.info(f"  Status: {operation.status} ({waited}s elapsed)")
            
            if not operation.done:
                logger.error(f"HeyGen timeout after {max_wait}s")
                return None
            
            if operation.status != "completed":
                logger.error(f"HeyGen failed: {operation.status}")
                return None
            
            # Step 5: Download video
            logger.info("Downloading video from HeyGen...")
            output_path = str(cached_path)
            video_path = self.heygen.download_video(operation.id, output_path)
            
            logger.info(f"Avatar video ready: {video_path}")
            return video_path
            
        except Exception as e:
            logger.error(f"Pure avatar generation failed: {e}")
            return None
    
    def _structure_script(self, script: dict) -> str:
        """Convert script dict to single narration string."""
        narration = f"{script['title']}\n\n"
        for section in script.get("sections", []):
            narration += section.get("text", "") + "\n\n"
        return narration.strip()
    
    def _get_cache_key(self, narration: str) -> str:
        """MD5 hash for cache lookup."""
        import hashlib
        return hashlib.md5(narration.encode()).hexdigest()[:12]
```

|-----------|--------|-----------|

| **Network dependency** | Download can timeout | Retry logic built-in |
| **Cost: $29/month** | Unlimited videos | Very cheap vs alternatives |
| **API rate limit** | Max 5 concurrent requests | Run single daily job (no issue) |

### `agents/avatar_agent.py` (Hybrid variant, key method)

```python
def generate_hybrid_avatar(self, script: dict, video_agent, run_id: str = None) -> str | None:
    """Generate video with avatar + Veo B-roll for each section."""
    try:
        sections = script.get("sections", [])
        visual_queries = script.get("visual_queries", [])
        
        if len(visual_queries) != len(sections):
            logger.warning(f"Mismatch: {len(sections)} sections but {len(visual_queries)} queries")
            visual_queries = visual_queries[:len(sections)]  # Truncate
        
        # Step 1: Generate avatar for each section (in parallel, simulated with queue)
        logger.info(f"Generating {len(sections)} avatar sections...")
        avatar_operations = []
        for i, section in enumerate(sections):
            narration = section.get("text", "")
            operation = self.heygen.create_video(
                avatar_id=config.HEYGEN_AVATAR_ID,
                script=narration,
                voice_params={"language": "en", "accent": "US", "speed": 1.0}
            )
            avatar_operations.append((i, operation.id))
            logger.info(f"  Queued avatar {i+1}/{len(sections)}")
        
        # Poll all avatar videos
        logger.info("Polling avatar generation...")
        avatar_clips = [None] * len(sections)
        for section_idx, operation_id in avatar_operations:
            operation = self._poll_until_done(operation_id, max_wait=600)
            if operation.done:
                avatar_path = self.heygen.download_video(operation_id, 
                    f"{self.cache_dir}/avatar_{section_idx}.mp4")
                avatar_clips[section_idx] = avatar_path
                logger.info(f"  Downloaded avatar {section_idx+1}")
        
        # Step 2: Generate B-roll (Veo) for each section
        logger.info(f"Generating {len(sections)} Veo B-rolls...")
        veo_clips = []
        for i, query in enumerate(visual_queries):
            veo_path = video_agent.gcp_veo_agent.generate(query, section_idx=i, duration=8)
            veo_clips.append(veo_path)
            logger.info(f"  Generated B-roll {i+1}/{len(sections)}")
        
        # Step 3: Composite avatar + B-roll
        logger.info("Compositing avatar + B-roll...")
        composite_clips = []
        for i, (avatar_path, veo_path) in enumerate(zip(avatar_clips, veo_clips)):
            if not avatar_path or not veo_path:
                logger.warning(f"Skipping section {i} (missing avatar or B-roll)")
                continue
            
            composite = self._composite_section(avatar_path, veo_path, section_idx=i)
            composite_clips.append(composite)
            logger.info(f"  Composited section {i+1}")
        
        # Step 4: Concatenate and add music
        logger.info("Assembling final video...")
        from moviepy import concatenate_videoclips, AudioFileClip
        from moviepy.audio.fx import MultiplyVolume
        
        final_video = concatenate_videoclips(composite_clips, method="chain")
        
        # Add background music
        music_path = self._select_music()
        music = AudioFileClip(music_path).with_duration(final_video.duration)
        final_video = final_video.with_audio(
            CompositeAudioClip([
                final_video.audio,
                music.with_effects([MultiplyVolume(0.15)])
            ])
        )
        
        # Save final
        output_path = f"{self.cache_dir}/hybrid_final.mp4"
        logger.info(f"Writing final video to {output_path}...")
        final_video.write_videofile(output_path, codec="libx264", fps=30)
        
        logger.info(f"Hybrid video ready: {output_path}")
        return output_path
        
    except Exception as e:
        logger.error(f"Hybrid generation failed: {e}")
        return None

def _composite_section(self, avatar_path: str, bg_path: str, section_idx: int) -> VideoClip:
    """Composite avatar + B-roll using MoviePy."""
    from moviepy import VideoFileClip, CompositeVideoClip
    
    avatar_clip = VideoFileClip(avatar_path)
    bg_clip = VideoFileClip(bg_path)
    
    # Stretch B-roll to match avatar duration
    bg_duration = avatar_clip.duration
    if bg_clip.duration < bg_duration:
        # Loop B-roll to fill duration
        loops = int(bg_duration / bg_clip.duration) + 1
        from moviepy import concatenate_videoclips
        bg_extended = concatenate_videoclips([bg_clip] * loops, method="chain")
        bg_clip = bg_extended.subclipped(0, bg_duration)
    
    # Position avatar as 60% screen (right side), B-roll as background
    avatar_scaled = avatar_clip.resized((1152, 1080))  # 60% of 1920
    bg_scaled = bg_clip.resized((1920, 1080))
    
    composite = CompositeVideoClip([
        bg_scaled.with_position("topleft"),
        avatar_scaled.with_position((768, 0))  # Right side
    ]).with_duration(bg_duration)
    
    return composite
```

```python
"""MoviePy compositing logic for hybrid videos."""
from moviepy import VideoFileClip, CompositeVideoClip, concatenate_videoclips

def composite_avatar_with_background(
    avatar_clip: VideoFileClip,
    bg_clip: VideoFileClip,
    position: str = "corner",
    opacity: float = 0.8
) -> VideoFileClip:
    """
    Composite avatar + background using MoviePy 2.x.
    
    Args:
        avatar_clip: Main avatar video
        bg_clip: Background B-roll
        position: "corner" (avatar 60% right) | "split" (avatar 50% left, bg 50% right)
        opacity: Alpha blending for avatar (0.0-1.0)
    
    Returns:
        Composite VideoClip
    """
    
    # Ensure same duration
    target_duration = avatar_clip.duration
    if bg_clip.duration < target_duration:
        # Loop B-roll
        loops = int(target_duration / bg_clip.duration) + 1
        bg_looped = concatenate_videoclips([bg_clip] * loops, method="chain")
        bg_clip = bg_looped.subclipped(0, target_duration)
    elif bg_clip.duration > target_duration:
        bg_clip = bg_clip.subclipped(0, target_duration)
    
    # Resize
    avatar_w, avatar_h = avatar_clip.size
    bg_w, bg_h = 1920, 1080
    
    if position == "corner":
        # Avatar 60% on right, B-roll full behind
        avatar_resized = avatar_clip.resized((1152, 1080))  # 60% of 1920
        bg_resized = bg_clip.resized((1920, 1080))
        
        composite = CompositeVideoClip([
            bg_resized.with_position((0, 0)),
            avatar_resized.with_position((768, 0)).with_opacity(opacity)
        ]).with_duration(target_duration)
    
    elif position == "split":
        # Avatar 50% left, B-roll 50% right
        avatar_resized = avatar_clip.resized((960, 1080))  # 50% of 1920
        bg_resized = bg_clip.resized((960, 1080))
        
        composite = CompositeVideoClip([
            avatar_resized.with_position((0, 0)),
            bg_resized.with_position((960, 0)).with_opacity(opacity)
        ]).with_duration(target_duration)
    
    return composite
```

## 3.5 Constraints & Limitations (Hybrid)

| Constraint | Impact | Mitigation |
|-----------|--------|-----------|
| **Total time: 2-3 hours** | Slow for daily uploads | Run overnight; use GitHub Actions scheduler |
| **API calls in parallel** | Could hit rate limits | Sequential fallback if needed |
| **Veo B-roll looping visible** | Repetition if section > 8s | Keep sections 6-8 min; B-roll cycles less |
| **MoviePy rendering slow** | 20-30 min to encode MP4 | Accept; run during off-peak |
| **Composite layout crowded** | Avatar + B-roll can feel busy | Choose minimal B-roll; don't use busy animations |
| **Veo B-roll quality varies** | Some prompts generate poorly | Improved prompts in avatar_prompts.py |
| **Audio sync issues** | Rare lip-sync problems on composite | MoviePy 2.x handles it well; occasional glitches OK |
| **Free trial constraint** | Veo only free for 90 days | After trial: switch to Ken Burns or accept $$ |
